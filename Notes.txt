// HOW TO INTERACT WITH POSTGRES
From the repo root (where docker-compose.yaml lives), run:
docker compose exec postgres psql -U inventory_user -d inventory_db

Then run SQL queries like:
SELECT * FROM inventory;
SELECT * FROM favorites;

// Powershell commands to interact with HTTPS endpoint
Invoke-WebRequest -Method Post "http://localhost:8080/create?sku=123&receiptAmount=5"
Invoke-WebRequest -Method Post "http://localhost:8080/purchase?sku=321&amount=4"    
Invoke-WebRequest -Method GET "http://localhost:8080/favorites?filter=least&count=2"
